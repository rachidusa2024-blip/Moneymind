  area.innerHTML = '<select class="lang-select" id="curSel"><option value="USD">USD — US Dollar</option><option value="CAD">CAD — Canadian Dollar</option><option value="EUR">EUR — Euro</option><option value="GBP">GBP — British Pound</option><option value="MXN">MXN — Mexican Peso</option><option value="BRL">BRL — Brazilian Real</option><option value="ARS">ARS — Argentine Peso</option><option value="MAD">MAD — Moroccan Dirham</option><option value="DZD">DZD — Algerian Dinar</option><option value="INR">INR — Indian Rupee</option><option value="PKR">PKR — Pakistani Rupee</option><option value="AED">AED — UAE Dirham</option><option value="TRY">TRY — Turkish Lira</option><option value="other">Other currency</option></select><button class="btn-continue" onclick="submitCurrency()">Continue →</button>';
  window._onCurSubmit = onSubmit;
}

function submitCurrency() {
  var cur = document.getElementById('curSel').value;
  addUserMessage(cur);
  answers.currency = cur;
  clearInput();
  if (window._onCurSubmit) window._onCurSubmit(cur);
}

function addUserMessage(text) {
  var conv = document.getElementById('conversation');
  var div = document.createElement('div');
  div.style.cssText = 'display:flex;flex-direction:column;align-items:flex-end;animation:up 0.3s ease forwards';
  div.innerHTML = '<div style="font-size:10px;font-weight:600;letter-spacing:.1em;text-transform:uppercase;color:var(--muted);margin-bottom:6px">You</div><div style="background:rgba(255,255,255,0.06);border:1px solid var(--border);border-radius:16px 4px 16px 16px;padding:12px 16px;font-size:14px;color:var(--white);max-width:85%;text-align:right">' + text + '</div>';
  conv.appendChild(div);
  conv.scrollTop = conv.scrollHeight;
}

async function runStep(n) {
  step = n;
  if (n < steps.length) setProgress(steps[n].progress);
  var cur = answers.currency || 'USD';

  if (n === 0) {
    await addMessage("Good day. I am <strong>Vera</strong> — your personal financial coach.<br><br>I have guided clients through debt elimination, wealth building, and financial transformation for decades. Everything you share with me is completely private.<br><br>Before we begin, what language do you prefer?");
    showLangCurrency(function() { runStep(1); });

  } else if (n === 1) {
    await addMessage("Thank you. And what currency do you use? This ensures every number I give you is accurate for your situation.");
    showCurrencySelect(function() { runStep(2); });

  } else if (n === 2) {
    await addMessage("Now let us build your complete financial picture. I need accurate numbers to give you accurate advice.<br><br>What is your total monthly income after taxes — including your salary and any other sources?");
    showNumberInput('e.g. 4500', function(val) { answers.income = val; runStep(3); });

  } else if (n === 3) {
    await addMessage("Understood. Now your largest fixed expense — <strong>rent or mortgage</strong>. What do you pay monthly?");
    showNumberInput('e.g. 1200', function(val) { answers.rent = val; runStep(4); });

  } else if (n === 4) {
    await addMessage("And your other monthly expenses combined — food, transport, utilities, phone, subscriptions, everything else. Give me your best estimate.");
    showNumberInput('e.g. 1800', function(val) { answers.otherExpenses = val; runStep(5); });

  } else if (n === 5) {
    await addMessage("Now the number most people avoid discussing. What is your <strong>total debt</strong> — credit cards, loans, car payments, student debt — everything combined?");
    showNumberInput('0 if none', function(val) { answers.debt = val; runStep(6); });

  } else if (n === 6) {
    await addMessage("And your current savings — the total amount you have set aside right now across all accounts?");
    showNumberInput('0 if none', function(val) { answers.savings = val; runStep(7); });

  } else if (n === 7) {
    await addMessage("How many people depend on you financially? This includes children, a spouse who does not work, or family members you support.");
    showChoices([
      { value: 0, label: 'Only myself' },
      { value: 1, label: '1 person' },
      { value: 2, label: '2 people' },
      { value: 3, label: '3 or more people' }
    ], function(val) { answers.dependents = val; runStep(8); });

  } else if (n === 8) {
    await addMessage("One more important question. What is your biggest financial fear right now?");
    showChoices([
      { value: 'debt', label: 'Never getting out of debt' },
      { value: 'paycheck', label: 'Living paycheck to paycheck forever' },
      { value: 'emergency', label: 'A financial emergency I cannot handle' },
      { value: 'family', label: 'Not being able to provide for my family' },
      { value: 'future', label: 'Not having enough when I am older' }
    ], function(val, label) { answers.biggestFear = label; runStep(9); });

  } else if (n === 9) {
    await addMessage("Last question. If you could achieve one financial goal in the next 12 months — what would it be?");
    showChoices([
      { value: 'debt', label: 'Eliminate or significantly reduce my debt' },
      { value: 'save', label: 'Build real savings for the first time' },
      { value: 'house', label: 'Save for a home down payment' },
      { value: 'emergency', label: 'Build a 3-month emergency fund' },
      { value: 'freedom', label: 'Stop worrying about money every month' }
    ], function(val, label) { answers.topGoal = label; runAnalysis(); });
  }
}

async function runAnalysis() {
  clearInput();
  document.getElementById('conversation').style.display = 'none';
  document.getElementById('analysisScreen').style.display = 'block';

  var totalExpenses = answers.rent + answers.otherExpenses;
  var available = answers.income - totalExpenses;
  var expRatio = answers.income > 0 ? totalExpenses / answers.income : 1;
  var debtToIncome = answers.income > 0 ? answers.debt / (answers.income * 12) : 0;
  var savingsMonths = totalExpenses > 0 ? answers.savings / totalExpenses : 0;
  var monthsToDebtFree = available > 0 && answers.debt > 0 ? Math.ceil(answers.debt / (available * 0.5)) : 0;

  var score = 0;
  if (expRatio <= 0.5) score += 30; else if (expRatio <= 0.65) score += 22; else if (expRatio <= 0.75) score += 15; else if (expRatio <= 0.85) score += 8;
  if (answers.debt === 0) score += 30; else if (debtToIncome <= 0.5) score += 22; else if (debtToIncome <= 1) score += 15; else if (debtToIncome <= 2) score += 8; else score += 2;
  if (savingsMonths >= 6) score += 25; else if (savingsMonths >= 3) score += 18; else if (savingsMonths >= 1) score += 10; else if (answers.savings > 0) score += 4;
  if (available > 0 && available / answers.income >= 0.2) score += 15; else if (available > 0) score += 8;

  var cur = answers.currency;

  var actions = [];
  if (available <= 0) {
    actions.push('Your expenses exceed your income. The first priority is cutting ' + cur + ' ' + Math.abs(Math.round(available)).toLocaleString() + ' per month from your spending.');
  } else {
    actions.push('You have ' + cur + ' ' + Math.round(available).toLocaleString() + ' available monthly. I recommend allocating ' + cur + ' ' + Math.round(available * 0.5).toLocaleString() + ' toward debt and ' + cur + ' ' + Math.round(available * 0.3).toLocaleString() + ' toward savings.');
  }
  if (answers.debt > 0 && monthsToDebtFree > 0) {
    actions.push('At the recommended payment rate, you can be debt-free in approximately ' + monthsToDebtFree + ' months.');
  }
  if (savingsMonths < 1) {
    actions.push('Your emergency fund needs immediate attention. Begin with a target of ' + cur + ' ' + Math.round(totalExpenses * 1).toLocaleString() + ' — one month of expenses.');
  }
  actions.push('Your top priority matches your goal: ' + answers.topGoal + '. I will build this into your weekly plan.');

  profile = {
    score: score,
    available: available,
    expRatio: expRatio,
    monthsToDebtFree: monthsToDebtFree,
    actions: actions,
    cur: cur
  };

  await saveProfile();
  setTimeout(function() { showResult(score, cur, available, monthsToDebtFree, actions); }, 2500);
}

async function saveProfile() {
  try {
    await sb.from('profiles').update({
      financial_challenge: answers.biggestFear,
      monthly_income: answers.income,
      currency: answers.currency,
      language: answers.language,
      onboarding_completed: true,
      updated_at: new Date().toISOString()
    }).eq('id', user.id);

    await sb.from('financial_data').upsert({
      user_id: user.id,
      health_score: profile.score || 0,
      total_debt: answers.debt,
      total_savings: answers.savings,
      monthly_budget: answers.income - answers.rent - answers.otherExpenses,
      emergency_fund: answers.savings,
      updated_at: new Date().toISOString()
    }, { onConflict: 'user_id' });
  } catch(e) { console.log('Save error:', e); }
}

function showResult(score, cur, available, monthsToDebtFree, actions) {
  document.getElementById('analysisScreen').style.display = 'none';
  var rs = document.getElementById('resultScreen');
  rs.style.display = 'block';

  var grade = score >= 75 ? 'Strong financial foundation' : score >= 55 ? 'Good — clear path forward' : score >= 35 ? 'Needs focused attention' : 'Immediate action required';
  var debtFreeText = monthsToDebtFree > 0 ? monthsToDebtFree + ' months' : answers.debt === 0 ? 'Debt free' : 'TBD';
  var availableColor = available >= 0 ? 'teal' : 'red';

  rs.innerHTML = '<div class="msg-sender" style="margin-bottom:16px">Vera\'s Assessment</div>' +
    '<div class="result-score">' +
      '<div class="result-score-label">Financial Health Score</div>' +
      '<div class="result-score-num">' + score + '<span class="result-score-max">/100</span></div>' +
      '<div class="result-score-grade">' + grade + '</div>' +
    '</div>' +
    '<div class="result-grid">' +
      '<div class="result-stat"><div class="result-stat-label">Monthly surplus</div><div class="result-stat-val ' + availableColor + '">' + cur + ' ' + Math.round(Math.abs(available)).toLocaleString() + (available < 0 ? ' deficit' : '') + '</div></div>' +
      '<div class="result-stat"><div class="result-stat-label">Total debt</div><div class="result-stat-val ' + (answers.debt > 0 ? 'red' : 'teal') + '">' + cur + ' ' + answers.debt.toLocaleString() + '</div></div>' +
      '<div class="result-stat"><div class="result-stat-label">Debt-free timeline</div><div class="result-stat-val gold">' + debtFreeText + '</div></div>' +
      '<div class="result-stat"><div class="result-stat-label">Current savings</div><div class="result-stat-val teal">' + cur + ' ' + answers.savings.toLocaleString() + '</div></div>' +
    '</div>' +
    '<div class="result-actions">' +
      '<div class="result-actions-title">Vera\'s immediate recommendations</div>' +
      actions.map(function(a) { return '<div class="result-action">' + a + '</div>'; }).join('') +
    '</div>' +
    '<button class="btn-enter" onclick="enterDashboard()">Enter my financial dashboard →</button>' +
    '<button class="skip-link" onclick="enterDashboard()">Save and continue later</button>';

  document.getElementById('progressFill').style.width = '100%';
}

function enterDashboard() {
  localStorage.setItem('vera_profile', JSON.stringify(answers));
  localStorage.setItem('vera_score', profile.score);
  window.location.href = '/dashboard.html';
}
