function calculateScore() {
  const income = parseFloat(document.getElementById('income').value) || 0;
  const expenses = parseFloat(document.getElementById('expenses').value) || 0;
  const savings = parseFloat(document.getElementById('savings').value) || 0;
  const totalDebt = parseFloat(document.getElementById('totalDebtInput').value) || 0;
  const emergency = parseFloat(document.getElementById('emergencyFund').value) || 0;
  const currency = document.getElementById('scoreCurrency').value;

  if (!income) { alert('Please enter your monthly income'); return; }

  // Budget score (25pts) - expense ratio
  const expenseRatio = income > 0 ? expenses / income : 1;
  let budgetScore = 0;
  if (expenseRatio <= 0.5) budgetScore = 25;
  else if (expenseRatio <= 0.6) budgetScore = 22;
  else if (expenseRatio <= 0.7) budgetScore = 18;
  else if (expenseRatio <= 0.8) budgetScore = 14;
  else if (expenseRatio <= 0.9) budgetScore = 10;
  else if (expenseRatio <= 1.0) budgetScore = 5;
  else budgetScore = 0;

  // Debt score (25pts) - debt to income ratio
  const debtRatio = income > 0 ? totalDebt / (income * 12) : 0;
  let debtScore = 0;
  if (totalDebt === 0) debtScore = 25;
  else if (debtRatio <= 0.5) debtScore = 22;
  else if (debtRatio <= 1) debtScore = 18;
  else if (debtRatio <= 2) debtScore = 12;
  else if (debtRatio <= 3) debtScore = 7;
  else debtScore = 3;

  // Savings score (25pts) - savings rate
  const savingsRate = income > 0 ? savings / income : 0;
  let savingsScore = 0;
  if (savingsRate >= 0.3) savingsScore = 25;
  else if (savingsRate >= 0.2) savingsScore = 22;
  else if (savingsRate >= 0.15) savingsScore = 18;
  else if (savingsRate >= 0.1) savingsScore = 14;
  else if (savingsRate >= 0.05) savingsScore = 9;
  else if (savingsRate > 0) savingsScore = 5;
  else savingsScore = 0;

  // Emergency fund score (25pts) - months covered
  const monthsCovered = expenses > 0 ? emergency / expenses : 0;
  let emergencyScore = 0;
  if (monthsCovered >= 6) emergencyScore = 25;
  else if (monthsCovered >= 4) emergencyScore = 20;
  else if (monthsCovered >= 3) emergencyScore = 16;
  else if (monthsCovered >= 2) emergencyScore = 11;
  else if (monthsCovered >= 1) emergencyScore = 7;
  else if (emergency > 0) emergencyScore = 3;

  const total = budgetScore + debtScore + savingsScore + emergencyScore;

  // Grade
  let grade, label, color;
  if (total >= 85) { grade = 'A+'; label = 'Excellent financial health'; color = '#10B981'; }
  else if (total >= 70) { grade = 'A'; label = 'Very good financial health'; color = '#10B981'; }
  else if (total >= 55) { grade = 'B'; label = 'Good — room to improve'; color = '#F59E0B'; }
  else if (total >= 40) { grade = 'C'; label = 'Fair — needs attention'; color = '#F97316'; }
  else if (total >= 25) { grade = 'D'; label = 'Needs significant improvement'; color = '#EF4444'; }
  else { grade = 'F'; label = 'Critical — action needed now'; color = '#EF4444'; }

  // Animate score ring
  const offset = 283 - (283 * total / 100);
  const arc = document.getElementById('scoreArc');
  arc.style.setProperty('--target-offset', offset);
  arc.style.stroke = color;
  arc.style.animation = 'none';
  arc.offsetHeight;
  arc.style.animation = 'scoreAnim 1.5s ease forwards';

  document.getElementById('scoreNum').textContent = total;
  document.getElementById('scoreGrade').textContent = 'Grade: ' + grade;
  document.getElementById('scoreLabel').textContent = label;
  document.getElementById('scoreBadges').innerHTML = `
    <div class="score-badge">Budget: ${budgetScore}/25</div>
    <div class="score-badge">Debt: ${debtScore}/25</div>
    <div class="score-badge">Savings: ${savingsScore}/25</div>
    <div class="score-badge">Emergency: ${emergencyScore}/25</div>
  `;

  // Show categories
  document.getElementById('categories').style.display = 'grid';
  updateCat('budget', budgetScore, expenseRatio <= 0.7 ? 'Your budget control is good!' : 'Reduce expenses to below 70% of income', expenseRatio <= 0.7 ? 'var(--green)' : 'var(--orange)');
  updateCat('debt', debtScore, totalDebt === 0 ? 'Amazing — zero debt!' : debtRatio <= 1 ? 'Debt level is manageable' : 'Focus on reducing debt aggressively', totalDebt === 0 ? 'var(--green)' : debtRatio <= 1 ? 'var(--gold)' : 'var(--red)');
  updateCat('savings', savingsScore, savingsRate >= 0.2 ? 'Excellent savings rate!' : 'Target saving 20% of income', savingsRate >= 0.2 ? 'var(--green)' : 'var(--orange)');
  updateCat('emergency', emergencyScore, monthsCovered >= 3 ? 'Emergency fund is solid' : `Build to ${Math.ceil((3 - monthsCovered) * expenses).toLocaleString()} more to cover 3 months`, monthsCovered >= 3 ? 'var(--green)' : 'var(--orange)');

  document.getElementById(`budgetBar`).style.width = (budgetScore / 25 * 100) + '%';
  document.getElementById(`debtBar`).style.width = (debtScore / 25 * 100) + '%';
  document.getElementById(`savingsBar`).style.width = (savingsScore / 25 * 100) + '%';
  document.getElementById(`emergencyBar`).style.width = (emergencyScore / 25 * 100) + '%';
  document.getElementById(`budgetScore`).textContent = budgetScore;
  document.getElementById(`debtScore`).textContent = debtScore;
  document.getElementById(`savingsScore`).textContent = savingsScore;
  document.getElementById(`emergencyScore`).textContent = emergencyScore;

  // Maya tips
  getMayaTips(total, budgetScore, debtScore, savingsScore, emergencyScore, currency, income, savings, totalDebt, emergency, expenses);
  document.getElementById('scoreHero').scrollIntoView({ behavior: 'smooth' });
}

function updateCat(id, score, tip, color) {
  document.getElementById(id + 'Tip').textContent = tip;
  document.getElementById(id + 'Tip').style.background = color + '15';
  document.getElementById(id + 'Tip').style.color = color;
}

async function getMayaTips(total, b, d, s, e, currency, income, savings, debt, emergency, expenses) {
  const tips = document.getElementById('mayaTips');
  const actions = document.getElementById('mtActions');
  const content = document.getElementById('mtContent');
  tips.style.display = 'block';

  const weakest = [
    {name:'Budget', score:b, max:25},
    {name:'Debt', score:d, max:25},
    {name:'Savings', score:s, max:25},
    {name:'Emergency', score:e, max:25}
  ].sort((a,b) => a.score - b.score);

  // Local tips
  const localTips = [];
  if (b < 18) localTips.push({title:'Cut expenses this month', desc:`Try reducing spending by ${currency} ${Math.round(income * 0.05).toLocaleString()} — that alone adds 3 points to your score.`});
  if (d < 18 && debt > 0) localTips.push({title:'Attack your highest debt', desc:`Pay ${currency} ${Math.round(Math.min(income * 0.1, debt)).toLocaleString()} extra on your most expensive debt this month.`});
  if (s < 18) localTips.push({title:'Boost your savings rate', desc:`Set up an automatic transfer of ${currency} ${Math.round(income * 0.05).toLocaleString()} on payday. You won't miss it.`});
  if (e < 18) localTips.push({title:'Build your emergency fund', desc:`Add ${currency} ${Math.round(expenses * 0.5).toLocaleString()} to your emergency fund — that gets you one step closer to 3 months covered.`});

  actions.innerHTML = localTips.slice(0,3).map(t => `<div class="mt-action"><strong>${t.title}</strong>${t.desc}</div>`).join('');

  try {
    const res = await fetch('/api/chat', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        messages: [{ role: 'user', content: `You are Maya. Financial health score: ${total}/100. Budget: ${b}/25, Debt: ${d}/25, Savings: ${s}/25, Emergency: ${e}/25. Income: ${currency} ${income}/mo. Give one encouraging sentence about their overall score and one key insight. Max 50 words. No emojis.` }],
        mode: 'coach'
      })
    });
    const data = await res.json();
    if (data.reply) content.innerHTML = data.reply.replace(/\*\*(.*?)\*\*/g,'<strong>$1</strong>');
  } catch(e) {
    content.innerHTML = total >= 70 ? 'You are doing well. Keep building on your strengths.' : 'Every point you add to your score represents real money saved and stress reduced. Focus on one category at a time.';
  }
}
